119 template<class SerialPort>
120 void MidiInterface<SerialPort>::send(MidiType inType,
121  DataByte inData1,
122  DataByte inData2,
123  Channel inChannel)
124 {
125  // Then test if channel is valid
126  if (inChannel >= MIDI_CHANNEL_OFF ||
127  inChannel == MIDI_CHANNEL_OMNI ||
128  inType < NoteOff)
129  {
130 
131 #if MIDI_USE_RUNNING_STATUS
132  mRunningStatus_TX = InvalidType;
133 #endif
134 
135  return; // Don't send anything
136  }
137 
138  if (inType <= PitchBend) // Channel messages
139  {
140  // Protection: remove MSBs on data
141  inData1 &= 0x7F;
142  inData2 &= 0x7F;
143 
144  const StatusByte status = getStatus(inType, inChannel);
145 
146 #if MIDI_USE_RUNNING_STATUS
147  // Check Running Status
148  if (mRunningStatus_TX != status)
149  {
150  // New message, memorise and send header
151  mRunningStatus_TX = status;
152  mSerial.write(mRunningStatus_TX);
153  }
154 #else
155  // Don't care about running status, send the status byte.
156  mSerial.write(status);
157 #endif
158 
159  // Then send data
160  mSerial.write(inData1);
161  if (inType != ProgramChange && inType != AfterTouchChannel)
162  mSerial.write(inData2);
163 
164  return;
165  }
166  else if (inType >= TuneRequest && inType <= SystemReset)
167  sendRealTime(inType); // System Real-time and 1 byte.
168 }

490 // Private method: MIDI parser
491 template<class SerialPort>
492 bool MidiInterface<SerialPort>::parse(Channel inChannel)
493 {
494  const byte bytes_available = mSerial.available();
495 
496  if (bytes_available == 0)
497  // No data available.
498  return false;
499 
500  // Parsing algorithm:
501  // Get a byte from the serial buffer.
502  // If there is no pending message to be recomposed, start a new one.
503  // - Find type and channel (if pertinent)
504  // - Look for other bytes in buffer, call parser recursively,
505  // until the message is assembled or the buffer is empty.
506  // Else, add the extracted byte to the pending message, and check validity.
507  // When the message is done, store it.
508 
509  const byte extracted = mSerial.read();
510 
511  if (mPendingMessageIndex == 0)
512  {
513  // Start a new pending message
514  mPendingMessage[0] = extracted;
515 
516  // Check for running status first
517  switch (getTypeFromStatusByte(mRunningStatus_RX))
518  {
519  // Only these types allow Running Status:
520  case NoteOff:
521  case NoteOn:
522  case AfterTouchPoly:
523  case ControlChange:
524  case ProgramChange:
525  case AfterTouchChannel:
526  case PitchBend:
527 
528  // If the status byte is not received, prepend it
529  // to the pending message
530  if (extracted < 0x80)
531  {
532  mPendingMessage[0] = mRunningStatus_RX;
533  mPendingMessage[1] = extracted;
534  mPendingMessageIndex = 1;
535  }
536  // Else: well, we received another status byte,
537  // so the running status does not apply here.
538  // It will be updated upon completion of this message.
539 
540  if (mPendingMessageIndex >= (mPendingMessageExpectedLenght-1))
541  {
542  mMessage.type = getTypeFromStatusByte(mPendingMessage[0]);
543  mMessage.channel = (mPendingMessage[0] & 0x0F)+1;
544  mMessage.data1 = mPendingMessage[1];
545 
546  // Save data2 only if applicable
547  if (mPendingMessageExpectedLenght == 3)
548  mMessage.data2 = mPendingMessage[2];
549  else
550  mMessage.data2 = 0;
551 
552  mPendingMessageIndex = 0;
553  mPendingMessageExpectedLenght = 0;
554  mMessage.valid = true;
555  return true;
556  }
557  break;
558  default:
559  // No running status
560  break;
561  }
562 
563 
564  switch (getTypeFromStatusByte(mPendingMessage[0]))
565  {
566  // 1 byte messages
567  case Start:
568  case Continue:
569  case Stop:
570  case Clock:
571  case ActiveSensing:
572  case SystemReset:
573  case TuneRequest:
574  // Handle the message type directly here.
575  mMessage.type = getTypeFromStatusByte(mPendingMessage[0]);
576  mMessage.channel = 0;
577  mMessage.data1 = 0;
578  mMessage.data2 = 0;
579  mMessage.valid = true;
580 
581  // \fix Running Status broken when receiving Clock messages.
582  // Do not reset all input attributes, Running Status must remain unchanged.
583  //resetInput();
584 
585  // We still need to reset these
586  mPendingMessageIndex = 0;
587  mPendingMessageExpectedLenght = 0;
588 
589  return true;
590  break;
591 
592  // 2 bytes messages
593  case ProgramChange:
594  case AfterTouchChannel:
595  case TimeCodeQuarterFrame:
596  case SongSelect:
597  mPendingMessageExpectedLenght = 2;
598  break;
599 
600  // 3 bytes messages
601  case NoteOn:
602  case NoteOff:
603  case ControlChange:
604  case PitchBend:
605  case AfterTouchPoly:
606  case SongPosition:
607  mPendingMessageExpectedLenght = 3;
608  break;
609 
610  case SystemExclusive:
611  // The message can be any lenght
612  // between 3 and MIDI_SYSEX_ARRAY_SIZE bytes
613  mPendingMessageExpectedLenght = MIDI_SYSEX_ARRAY_SIZE;
614  mRunningStatus_RX = InvalidType;
615  break;
616 
617  case InvalidType:
618  default:
619  // This is obviously wrong. Let's get the hell out'a here.
620  resetInput();
621  return false;
622  break;
623  }
624 
625  // Then update the index of the pending message.
626  mPendingMessageIndex++;
627 
628 #if USE_1BYTE_PARSING
629  // Message is not complete.
630  return false;
631 #else
632  // Call the parser recursively
633  // to parse the rest of the message.
634  return parse(inChannel);
635 #endif
636 
637  }
638  else
639  {
640  // First, test if this is a status byte
641  if (extracted >= 0x80)
642  {
643  // Reception of status bytes in the middle of an uncompleted message
644  // are allowed only for interleaved Real Time message or EOX
645  switch (extracted)
646  {
647  case Clock:
648  case Start:
649  case Continue:
650  case Stop:
651  case ActiveSensing:
652  case SystemReset:
653 
654  // Here we will have to extract the one-byte message,
655  // pass it to the structure for being read outside
656  // the MIDI class, and recompose the message it was
657  // interleaved into. Oh, and without killing the running status..
658  // This is done by leaving the pending message as is,
659  // it will be completed on next calls.
660 
661  mMessage.type = (MidiType)extracted;
662  mMessage.data1 = 0;
663  mMessage.data2 = 0;
664  mMessage.channel = 0;
665  mMessage.valid = true;
666  return true;
667 
668  break;
669 
670  // End of Exclusive
671  case 0xF7:
672  if (getTypeFromStatusByte(mPendingMessage[0]) == SystemExclusive)
673  {
674  // Store System Exclusive array in midimsg structure
675  for (byte i=0;i<MIDI_SYSEX_ARRAY_SIZE;i++)
676  mMessage.sysex_array[i] = mPendingMessage[i];
677 
678  mMessage.type = SystemExclusive;
679 
680  // Get length
681  mMessage.data1 = (mPendingMessageIndex+1) & 0xFF;
682  mMessage.data2 = (mPendingMessageIndex+1) >> 8;
683  mMessage.channel = 0;
684  mMessage.valid = true;
685 
686  resetInput();
687  return true;
688  }
689  else
690  {
691  // Well well well.. error.
692  resetInput();
693  return false;
694  }
695 
696  break;
697  default:
698  break;
699  }
700  }
701 
702  // Add extracted data byte to pending message
703  mPendingMessage[mPendingMessageIndex] = extracted;
704 
705  // Now we are going to check if we have reached the end of the message
706  if (mPendingMessageIndex >= (mPendingMessageExpectedLenght-1))
707  {
708  // "FML" case: fall down here with an overflown SysEx..
709  // This means we received the last possible data byte that can fit
710  // the buffer. If this happens, try increasing MIDI_SYSEX_ARRAY_SIZE.
711  if (getTypeFromStatusByte(mPendingMessage[0]) == SystemExclusive)
712  {
713  resetInput();
714  return false;
715  }
716 
717  mMessage.type = getTypeFromStatusByte(mPendingMessage[0]);
718  // Don't check if it is a Channel Message
719  mMessage.channel = (mPendingMessage[0] & 0x0F)+1;
720 
721  mMessage.data1 = mPendingMessage[1];
722 
723  // Save data2 only if applicable
724  if (mPendingMessageExpectedLenght == 3)
725  mMessage.data2 = mPendingMessage[2];
726  else
727  mMessage.data2 = 0;
728 
729  // Reset local variables
730  mPendingMessageIndex = 0;
731  mPendingMessageExpectedLenght = 0;
732 
733  mMessage.valid = true;
734 
735  // Activate running status (if enabled for the received type)
736  switch (mMessage.type)
737  {
738  case NoteOff:
739  case NoteOn:
740  case AfterTouchPoly:
741  case ControlChange:
742  case ProgramChange:
743  case AfterTouchChannel:
744  case PitchBend:
745  // Running status enabled: store it from received message
746  mRunningStatus_RX = mPendingMessage[0];
747  break;
748 
749  default:
750  // No running status
751  mRunningStatus_RX = InvalidType;
752  break;
753  }
754  return true;
755  }
756  else
757  {
758  // Then update the index of the pending message.
759  mPendingMessageIndex++;
760 
761 #if USE_1BYTE_PARSING
762  // Message is not complete.
763  return false;
764 #else
765  // Call the parser recursively
766  // to parse the rest of the message.
767  return parse(inChannel);
768 #endif
769  }
770  }
771 
772  // What are our chances to fall here?
773  return false;
774 }

983 // Private - launch callback function based on received type.
984 template<class SerialPort>
985 void MidiInterface<SerialPort>::launchCallback()
986 {
987  // The order is mixed to allow frequent messages to trigger their callback faster.
988  switch (mMessage.type)
989  {
990  // Notes
991  case NoteOff: if (mNoteOffCallback != 0) mNoteOffCallback(mMessage.channel,mMessage.data1,mMessage.data2); break;
992  case NoteOn: if (mNoteOnCallback != 0) mNoteOnCallback(mMessage.channel,mMessage.data1,mMessage.data2); break;
993 
994  // Real-time messages
995  case Clock: if (mClockCallback != 0) mClockCallback(); break;
996  case Start: if (mStartCallback != 0) mStartCallback(); break;
997  case Continue: if (mContinueCallback != 0) mContinueCallback(); break;
998  case Stop: if (mStopCallback != 0) mStopCallback(); break;
999  case ActiveSensing: if (mActiveSensingCallback != 0) mActiveSensingCallback(); break;
1000 
1001  // Continuous controllers
1002  case ControlChange: if (mControlChangeCallback != 0) mControlChangeCallback(mMessage.channel,mMessage.data1,mMessage.data2); break;
1003  case PitchBend: if (mPitchBendCallback != 0) mPitchBendCallback(mMessage.channel,(int)((mMessage.data1 & 0x7F) | ((mMessage.data2 & 0x7F)<< 7)) + MIDI_PITCHBEND_MIN); break; // TODO: check this
1004  case AfterTouchPoly: if (mAfterTouchPolyCallback != 0) mAfterTouchPolyCallback(mMessage.channel,mMessage.data1,mMessage.data2); break;
1005  case AfterTouchChannel: if (mAfterTouchChannelCallback != 0) mAfterTouchChannelCallback(mMessage.channel,mMessage.data1); break;
1006 
1007  case ProgramChange: if (mProgramChangeCallback != 0) mProgramChangeCallback(mMessage.channel,mMessage.data1); break;
1008  case SystemExclusive: if (mSystemExclusiveCallback != 0) mSystemExclusiveCallback(mMessage.sysex_array,mMessage.data1); break;
1009 
1010  // Occasional messages
1011  case TimeCodeQuarterFrame: if (mTimeCodeQuarterFrameCallback != 0) mTimeCodeQuarterFrameCallback(mMessage.data1); break;
1012  case SongPosition: if (mSongPositionCallback != 0) mSongPositionCallback((mMessage.data1 & 0x7F) | ((mMessage.data2 & 0x7F)<< 7)); break;
1013  case SongSelect: if (mSongSelectCallback != 0) mSongSelectCallback(mMessage.data1); break;
1014  case TuneRequest: if (mTuneRequestCallback != 0) mTuneRequestCallback(); break;
1015 
1016  case SystemReset: if (mSystemResetCallback != 0) mSystemResetCallback(); break;
1017  case InvalidType:
1018  default:
1019  break;
1020  }

1083 // This method is called upon reception of a message
1084 // and takes care of Thru filtering and sending.
1085 template<class SerialPort>
1086 void MidiInterface<SerialPort>::thruFilter(Channel inChannel)
1087 {
1088 
1089  /*
1090  This method handles Soft-Thru filtering.
1091 
1092  Soft-Thru filtering:
1093  - All system messages (System Exclusive, Common and Real Time) are passed to output unless filter is set to Off
1094  - Channel messages are passed to the output whether their channel is matching the input channel and the filter setting
1095 
1096  */
1097 
1098  // If the feature is disabled, don't do anything.
1099  if (!mThruActivated || (mThruFilterMode == Off))
1100  return;
1101 
1102 
1103  // First, check if the received message is Channel
1104  if (mMessage.type >= NoteOff && mMessage.type <= PitchBend)
1105  {
1106  const bool filter_condition = ((mMessage.channel == mInputChannel) ||
1107  (mInputChannel == MIDI_CHANNEL_OMNI));
1108 
1109  // Now let's pass it to the output
1110  switch (mThruFilterMode)
1111  {
1112  case Full:
1113  send(mMessage.type,
1114  mMessage.data1,
1115  mMessage.data2,
1116  mMessage.channel);
1117  return;
1118  break;
1119  case SameChannel:
1120  if (filter_condition)
1121  {
1122  send(mMessage.type,
1123  mMessage.data1,
1124  mMessage.data2,
1125  mMessage.channel);
1126  return;
1127  }
1128  break;
1129  case DifferentChannel:
1130  if (!filter_condition)
1131  {
1132  send(mMessage.type,
1133  mMessage.data1,
1134  mMessage.data2,
1135  mMessage.channel);
1136  return;
1137  }
1138  break;
1139  case Off:
1140  // Do nothing.
1141  // Technically it's impossible to get there because
1142  // the case was already tested earlier.
1143  break;
1144  default:
1145  break;
1146  }
1147  }
1148  else
1149  {
1150  // Send the message to the output
1151  switch (mMessage.type)
1152  {
1153  // Real Time and 1 byte
1154  case Clock:
1155  case Start:
1156  case Stop:
1157  case Continue:
1158  case ActiveSensing:
1159  case SystemReset:
1160  case TuneRequest:
1161  sendRealTime(mMessage.type);
1162  return;
1163  break;
1164 
1165  case SystemExclusive:
1166  // Send SysEx (0xF0 and 0xF7 are included in the buffer)
1167  sendSysEx(mMessage.data1,mMessage.sysex_array,true);
1168  return;
1169  break;
1170 
1171  case SongSelect:
1172  sendSongSelect(mMessage.data1);
1173  return;
1174  break;
1175 
1176  case SongPosition:
1177  sendSongPosition(mMessage.data1 | ((unsigned)mMessage.data2<<7));
1178  return;
1179  break;
1180 
1181  case TimeCodeQuarterFrame:
1182  sendTimeCodeQuarterFrame(mMessage.data1,mMessage.data2);
1183  return;
1184  break;
1185  default:
1186  break;
1187  }
1188  }
1189 }
